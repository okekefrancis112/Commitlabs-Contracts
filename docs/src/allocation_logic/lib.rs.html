116    pub fn initialize(env: Env, admin: Address, commitment_core: Address) -> Result<(), Error> {
117        // Check if already initialized
118        if env.storage().instance().has(&DataKey::Initialized) {
119            return Err(Error::AlreadyInitialized);
120        }
121        
122        // Validate addresses
123        admin.require_auth();
124        
125        // Set storage
126        env.storage().instance().set(&DataKey::Admin, &admin);
127        env.storage().instance().set(&DataKey::CommitmentCore, &commitment_core);
128        env.storage().instance().set(&DataKey::Initialized, &true);
129        env.storage().instance().set(&DataKey::PoolRegistry, &Vec::<u32>::new(&env));
130        
131        // Emit initialization event
132        env.events().publish(
133            (symbol_short!("init"), symbol_short!("alloc")),
134            admin
135        );
136        
137        Ok(())
138    }

144    pub fn register_pool(
145        env: Env,
146        admin: Address,
147        pool_id: u32,
148        risk_level: RiskLevel,
149        apy: u32,
150        max_capacity: i128,
151    ) -> Result<(), Error> {
152        admin.require_auth();
153        Self::require_initialized(&env)?;
154        Self::require_admin(&env, &admin)?;
155        Self::require_no_reentrancy(&env)?;
156
157        // Input validation
158        if max_capacity <= 0 {
159            return Err(Error::InvalidCapacity);
160        }
161        
162        if apy > 100_000 {  // Max 1000% APY (10000 basis points = 100%)
163            return Err(Error::InvalidAPY);
164        }
165
166        // Check if pool already exists
167        if env.storage().persistent().has(&DataKey::Pool(pool_id)) {
168            return Err(Error::InvalidPoolId);
169        }
170
171        let pool = Pool {
172            pool_id,
173            risk_level,
174            apy,
175            total_liquidity: 0,
176            max_capacity,
177            active: true,
178            created_at: env.ledger().timestamp(),
179            updated_at: env.ledger().timestamp(),
180        };
181
182        // Store pool
183        env.storage().persistent().set(&DataKey::Pool(pool_id), &pool);
184        
185        // Add to registry
186        let mut registry: Vec<u32> = env.storage().instance()
187            .get(&DataKey::PoolRegistry)
188            .unwrap_or(Vec::new(&env));
189        registry.push_back(pool_id);
190        env.storage().instance().set(&DataKey::PoolRegistry, &registry);
191
192        // Emit event
193        env.events().publish(
194            (symbol_short!("pool_reg"), pool_id),
195            risk_level
196        );
197
198        Ok(())
199    }

259    /// Allocate funds according to strategy
260    /// 
261    /// # Formal Verification
262    /// **Preconditions:**
263    /// - Contract is initialized
264    /// - `amount > 0`
265    /// - `reentrancy_guard == false`
266    /// - No existing allocation for `commitment_id`
267    /// 
268    /// **Postconditions:**
269    /// - `get_allocation(commitment_id).total_allocated == amount`
270    /// - For all pools P: `P.total_liquidity <= P.max_capacity`
271    /// - `reentrancy_guard == false`
272    /// 
273    /// **Invariants Maintained:**
274    /// - INV-4: Reentrancy guard invariant
275    /// - Pool capacity never exceeded
276    /// 
277    /// **Security Properties:**
278    /// - SP-1: Reentrancy protection
279    /// - SP-3: Arithmetic safety (overflow checks)
280    pub fn allocate(
281        env: Env,
282        caller: Address,
283        commitment_id: u64,
284        amount: i128,
285        strategy: Strategy,
286    ) -> Result<AllocationSummary, Error> {
287        caller.require_auth();
288        Self::require_initialized(&env)?;
289        Self::require_no_reentrancy(&env)?;
290
291        // Rate limit allocations per caller address
292        let fn_symbol = symbol_short!("alloc");
293        RateLimiter::check(&env, &caller, &fn_symbol);
294
295        // Set reentrancy guard
296        Self::set_reentrancy_guard(&env, true);
297
298        // Input validation
299        if amount <= 0 {
300            Self::set_reentrancy_guard(&env, false);
301            return Err(Error::InvalidAmount);
302        }
303
304        // Check for existing allocation (prevent double allocation)
305        if env.storage().persistent().has(&DataKey::Allocations(commitment_id)) {
306            Self::set_reentrancy_guard(&env, false);
307            return Err(Error::AlreadyInitialized);
308        }
309
310        // Store allocation ownership
311        env.storage().persistent().set(&DataKey::AllocationOwner(commitment_id), &caller);
312        
313        // Store the strategy
314        env.storage().persistent().set(&DataKey::Strategy(commitment_id), &strategy);
315
316        // Get pools based on strategy
317        let pools = Self::select_pools(&env, strategy)?;
318        
319        if pools.is_empty() {
320            Self::set_reentrancy_guard(&env, false);
321            return Err(Error::NoSuitablePools);
322        }
323
324        // Calculate allocation amounts with overflow protection
325        let allocation_plan = Self::calculate_allocation(&env, amount, &pools, strategy)?;
326        
327        // Execute allocations
328        let mut allocations = Vec::new(&env);
329        let mut total_allocated = 0i128;
330        
331        for (pool_id, alloc_amount) in allocation_plan.iter() {
332            if alloc_amount <= 0 {
333                continue;
334            }
335
336            // Update pool liquidity with overflow check
337            let mut pool = Self::get_pool_internal(&env, pool_id)?;
338            
339            // Check pool is active
340            if !pool.active {
341                Self::set_reentrancy_guard(&env, false);
342                return Err(Error::PoolInactive);
343            }
344
345            // Safe addition with overflow check
346            let new_liquidity = pool.total_liquidity.checked_add(alloc_amount)
347                .ok_or(Error::ArithmeticOverflow)?;
348            
349            if new_liquidity > pool.max_capacity {
350                Self::set_reentrancy_guard(&env, false);
351                return Err(Error::PoolCapacityExceeded);
352            }
353            
354            pool.total_liquidity = new_liquidity;
355            pool.updated_at = env.ledger().timestamp();
356            env.storage().persistent().set(&DataKey::Pool(pool_id), &pool);
357
358            // Record allocation
359            let allocation = Allocation {
360                commitment_id,
361                pool_id,
362                amount: alloc_amount,
363                timestamp: env.ledger().timestamp(),
364            };
365            
366            allocations.push_back(allocation);
367            
368            // Safe addition
369            total_allocated = total_allocated.checked_add(alloc_amount)
370                .ok_or(Error::ArithmeticOverflow)?;
371        }
372
373        // Verify total matches requested amount
374        if total_allocated != amount {
375            Self::set_reentrancy_guard(&env, false);
376            return Err(Error::ArithmeticOverflow);
377        }
378
379        // Store allocations
380        env.storage().persistent().set(&DataKey::Allocations(commitment_id), &allocations);
381        env.storage().persistent().set(&DataKey::TotalAllocated(commitment_id), &total_allocated);
382
383        // Clear reentrancy guard
384        Self::set_reentrancy_guard(&env, false);
385
386        // Emit event
387        env.events().publish(
388            (symbol_short!("allocate"), commitment_id),
389            (strategy, amount)
390        );
391
392        Ok(AllocationSummary {
393            commitment_id,
394            strategy,
395            total_allocated,
396            allocations,
397        })
398    }

400    pub fn rebalance(
401        env: Env,
402        caller: Address,
403        commitment_id: u64
404    ) -> Result<AllocationSummary, Error> {
405        caller.require_auth();
406        Self::require_initialized(&env)?;
407        Self::require_no_reentrancy(&env)?;
408
409        // Rate limit rebalancing per caller address
410        let fn_symbol = symbol_short!("rebal");
411        RateLimiter::check(&env, &caller, &fn_symbol);
412
413        // Verify ownership
414        let owner: Address = env.storage().persistent()
415            .get(&DataKey::AllocationOwner(commitment_id))
416            .ok_or(Error::AllocationNotFound)?;
417        
418        if owner != caller {
419            return Err(Error::Unauthorized);
420        }
421
422        Self::set_reentrancy_guard(&env, true);
423
424        // Get current allocations
425        let current_allocations: Vec<Allocation> = env.storage().persistent()
426            .get(&DataKey::Allocations(commitment_id))
427            .ok_or(Error::AllocationNotFound)?;
428
429        // Get strategy
430        let strategy: Strategy = env.storage().persistent()
431            .get(&DataKey::Strategy(commitment_id))
432            .ok_or(Error::AllocationNotFound)?;
433
434        let mut total_amount = 0i128;
435        
436        // Remove old allocations from pools with overflow protection
437        for allocation in current_allocations.iter() {
438            total_amount = total_amount.checked_add(allocation.amount)
439                .ok_or(Error::ArithmeticOverflow)?;
440            
441            let mut pool = Self::get_pool_internal(&env, allocation.pool_id)?;
442            pool.total_liquidity = pool.total_liquidity.checked_sub(allocation.amount)
443                .ok_or(Error::ArithmeticOverflow)?;
444            pool.updated_at = env.ledger().timestamp();
445            env.storage().persistent().set(&DataKey::Pool(allocation.pool_id), &pool);
446        }
447
448        // Reallocate with current strategy
449        let pools = Self::select_pools(&env, strategy)?;
450        let allocation_plan = Self::calculate_allocation(&env, total_amount, &pools, strategy)?;
451        
452        let mut new_allocations = Vec::new(&env);
453        let mut new_total = 0i128;
454        
455        for (pool_id, alloc_amount) in allocation_plan.iter() {
456            if alloc_amount <= 0 {
457                continue;
458            }
459
460            let mut pool = Self::get_pool_internal(&env, pool_id)?;
461            
462            if !pool.active {
463                continue; // Skip inactive pools during rebalancing
464            }
465
466            let new_liquidity = pool.total_liquidity.checked_add(alloc_amount)
467                .ok_or(Error::ArithmeticOverflow)?;
468
469            if new_liquidity <= pool.max_capacity {
470                pool.total_liquidity = new_liquidity;
471                pool.updated_at = env.ledger().timestamp();
472                env.storage().persistent().set(&DataKey::Pool(pool_id), &pool);
473
474                let allocation = Allocation {
475                    commitment_id,
476                    pool_id,
477                    amount: alloc_amount,
478                    timestamp: env.ledger().timestamp(),
479                };
480                
481                new_allocations.push_back(allocation);
482                new_total = new_total.checked_add(alloc_amount)
483                    .ok_or(Error::ArithmeticOverflow)?;
484            }
485        }
486
487        env.storage().persistent().set(&DataKey::Allocations(commitment_id), &new_allocations);
488        env.storage().persistent().set(&DataKey::TotalAllocated(commitment_id), &new_total);
489
490        Self::set_reentrancy_guard(&env, false);
491
492        env.events().publish(
493            (symbol_short!("rebalance"), commitment_id),
494            new_total
495        );
496
497        Ok(AllocationSummary {
498            commitment_id,
499            strategy,
500            total_allocated: new_total,
501            allocations: new_allocations,
502        })
503    }

630    fn select_pools(env: &Env, strategy: Strategy) -> Result<Vec<Pool>, Error> {
631        let mut pools = Vec::new(env);
632        
633        let registry: Vec<u32> = env.storage().instance()
634            .get(&DataKey::PoolRegistry)
635            .unwrap_or(Vec::new(env));
636
637        for pool_id in registry.iter() {
638            if let Ok(pool) = Self::get_pool_internal(env, pool_id) {
639                if !pool.active {
640                    continue;
641                }
642
643                let include = match strategy {
644                    Strategy::Safe => matches!(pool.risk_level, RiskLevel::Low),
645                    Strategy::Balanced => true,
646                    Strategy::Aggressive => matches!(pool.risk_level, RiskLevel::High | RiskLevel::Medium),
647                };
648
649                if include {
650                    pools.push_back(pool);
651                }
652            }
653        }
654
655        Ok(pools)
656    }

658    fn calculate_allocation(
659        env: &Env,
660        total_amount: i128,
661        pools: &Vec<Pool>,
662        strategy: Strategy,
663    ) -> Result<Map<u32, i128>, Error> {
664        let mut allocation_map = Map::new(env);
665        let pool_count = pools.len();
666
667        if pool_count == 0 {
668            return Ok(allocation_map);
669        }
670
671        match strategy {
672            Strategy::Safe => {
673                let amount_per_pool = total_amount / pool_count as i128;
674                for pool in pools.iter() {
675                    allocation_map.set(pool.pool_id, amount_per_pool);
676                }
677            },
678            Strategy::Balanced => {
679                let mut low_risk_pools = Vec::new(env);
680                let mut medium_risk_pools = Vec::new(env);
681                let mut high_risk_pools = Vec::new(env);
682
683                for pool in pools.iter() {
684                    match pool.risk_level {
685                        RiskLevel::Low => low_risk_pools.push_back(pool),
686                        RiskLevel::Medium => medium_risk_pools.push_back(pool),
687                        RiskLevel::High => high_risk_pools.push_back(pool),
688                    }
689                }
690
691                // Safe percentage calculations with checked operations
692                let low_amount = total_amount.checked_mul(40)
693                    .and_then(|x| x.checked_div(100))
694                    .ok_or(Error::ArithmeticOverflow)?;
695                
696                let medium_amount = total_amount.checked_mul(40)
697                    .and_then(|x| x.checked_div(100))
698                    .ok_or(Error::ArithmeticOverflow)?;
699                
700                let high_amount = total_amount.checked_mul(20)
701                    .and_then(|x| x.checked_div(100))
702                    .ok_or(Error::ArithmeticOverflow)?;
703
704                Self::distribute_to_pools(env, &mut allocation_map, &low_risk_pools, low_amount)?;
705                Self::distribute_to_pools(env, &mut allocation_map, &medium_risk_pools, medium_amount)?;
706                Self::distribute_to_pools(env, &mut allocation_map, &high_risk_pools, high_amount)?;
707            },
708            Strategy::Aggressive => {
709                let mut medium_risk_pools = Vec::new(env);
710                let mut high_risk_pools = Vec::new(env);
711
712                for pool in pools.iter() {
713                    match pool.risk_level {
714                        RiskLevel::Medium => medium_risk_pools.push_back(pool),
715                        RiskLevel::High => high_risk_pools.push_back(pool),
716                        _ => {},
717                    }
718                }
719
720                let high_amount = total_amount.checked_mul(70)
721                    .and_then(|x| x.checked_div(100))
722                    .ok_or(Error::ArithmeticOverflow)?;
723                
724                let medium_amount = total_amount.checked_mul(30)
725                    .and_then(|x| x.checked_div(100))
726                    .ok_or(Error::ArithmeticOverflow)?;
727
728                Self::distribute_to_pools(env, &mut allocation_map, &high_risk_pools, high_amount)?;
729                Self::distribute_to_pools(env, &mut allocation_map, &medium_risk_pools, medium_amount)?;
730            },
731        }
732
733        Ok(allocation_map)
734    }

736    fn distribute_to_pools(
737        _env: &Env,
738        allocation_map: &mut Map<u32, i128>,
739        pools: &Vec<Pool>,
740        amount: i128,
741    ) -> Result<(), Error> {
742        let pool_count = pools.len();
743        if pool_count == 0 {
744            return Ok(());
745        }
746
747        let amount_per_pool = amount / pool_count as i128;
748        
749        for pool in pools.iter() {
750            let available_capacity = pool.max_capacity.checked_sub(pool.total_liquidity)
751                .ok_or(Error::ArithmeticOverflow)?;
752            
753            let alloc_amount = if amount_per_pool > available_capacity {
754                available_capacity
755            } else {
756                amount_per_pool
757            };
758
759            if alloc_amount > 0 {
760                allocation_map.set(pool.pool_id, alloc_amount);
761            }
762        }
763
764        Ok(())
765    }
766}
