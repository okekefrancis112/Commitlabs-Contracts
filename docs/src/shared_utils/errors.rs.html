1//! Error handling utilities and common error patterns
2
3use soroban_sdk::{Env, log};
4
5/// Error helper functions
6pub struct ErrorHelper;
7
8impl ErrorHelper {
9    /// Log an error message
10    /// 
11    /// # Arguments
12    /// * `e` - The environment
13    /// * `message` - The error message
14    pub fn log_error(e: &Env, message: &str) {
15        log!(e, "Error: {}", message);
16    }
17
18    /// Log an error with context
19    /// 
20    /// # Arguments
21    /// * `e` - The environment
22    /// * `context` - The error context
23    /// * `message` - The error message
24    pub fn log_error_with_context(e: &Env, context: &str, message: &str) {
25        log!(e, "Error [{}]: {}", context, message);
26    }
27
28    /// Panic with a formatted error message
29    /// 
30    /// # Arguments
31    /// * `e` - The environment
32    /// * `message` - The error message
33    /// 
34    /// # Panics
35    /// Always panics with the error message
36    pub fn panic_with_log(e: &Env, message: &str) -> ! {
37        Self::log_error(e, message);
38        panic!("{}", message);
39    }
40
41    /// Panic with context and formatted error message
42    /// 
43    /// # Arguments
44    /// * `e` - The environment
45    /// * `context` - The error context
46    /// * `message` - The error message
47    /// 
48    /// # Panics
49    /// Always panics with the formatted error message
50    pub fn panic_with_context(e: &Env, context: &str, message: &str) -> ! {
51        Self::log_error_with_context(e, context, message);
52        panic!("[{}] {}", context, message);
53    }
54
55    /// Require a condition to be true, panic otherwise
56    /// 
57    /// # Arguments
58    /// * `e` - The environment
59    /// * `condition` - The condition to check
60    /// * `message` - The error message if condition is false
61    /// 
62    /// # Panics
63    /// Panics with the error message if condition is false
64    pub fn require(e: &Env, condition: bool, message: &str) {
65        if !condition {
66            Self::panic_with_log(e, message);
67        }
68    }
69
70    /// Require a condition with context
71    /// 
72    /// # Arguments
73    /// * `e` - The environment
74    /// * `condition` - The condition to check
75    /// * `context` - The error context
76    /// * `message` - The error message if condition is false
77    /// 
78    /// # Panics
79    /// Panics with the formatted error message if condition is false
80    pub fn require_with_context(e: &Env, condition: bool, context: &str, message: &str) {
81        if !condition {
82            Self::panic_with_context(e, context, message);
83        }
84    }
85}
